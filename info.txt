
instalando o Bootstrap Via NPM

# Instalo Bootstrap
npm install Bootstrap


# Dependencias

npm install jquery

npm install popper.js

installo rubi e o sass...

#como compilar scss em css em tempo real

sass --watch node_modules/bootstrap/scss:node_modules/bootstrap/compiler

#compilar font-awesome junto

sass --watch node_modules/bootstrap/scss:node_modules/bootstrap/compiler node_modules/font-awesome/scss:node_modules/font-awesome/css view/sass:view/css
